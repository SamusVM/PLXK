CASE 			
	WHEN NEW.mark_id = 1 THEN	-- ініційовано
		BEGIN			
			CALL edms_add_demand_manage (NEW.document_id, NEW.mark_id, NEW.employee_seat_id); -- направляємо документ далі
		END;	
	WHEN NEW.mark_id IN (2, 3, 5, 6, 8, 10, 11) THEN -- погоджено, відмовлено, на доопрацювання, не заперечую, резолюція, ознайомлений, виконано
		BEGIN		
			CALL edms_add_demand_manage (NEW.document_id, NEW.mark_id, NEW.employee_seat_id); -- направляємо документ далі
			CALL edms_del_demand (NEW.document_id, NEW.employee_seat_id); -- видаляємо документ із flow користувача
		END;
	WHEN NEW.mark_id = 7 THEN -- закриття
		BEGIN
			UPDATE edms_document -- закриваємо сам документ
			SET closed = TRUE
			WHERE id = NEW.document_id;
		
			CALL edms_del_demand_all (NEW.document_id); -- видаляємо документ із усіх flow
		END;		
	ELSE	-- коментар або файл - не робимо нічого
		BEGIN
		END;			
END CASE
